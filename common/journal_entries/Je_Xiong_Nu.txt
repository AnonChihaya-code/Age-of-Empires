je_loong_city_revival = { #龙城复兴
	icon = "gfx/interface/icons/law_icons/presidential_republic.dds" #总统共和制
	group = je_group_historical_content
	modifiers_while_active = {
		je_loong_city_revival_modifier
	}
	immediate = {
		set_variable = {
			name = ai_je_loong_city_revival
			value = 0
		}
	}
	complete = {
		trigger_if = {
			limit = {
				is_player = yes
			}
			army_size >= 100
			any_scope_state = {
				state_region = s:STATE_URGA
				turmoil <= 0.05
				has_famine = no
				any_scope_building = {
					is_building_type = building_urban_center
					level >= 10
				}
				any_scope_building = {
					is_building_type = building_university
					level >= 5
					occupancy >= 0.9
					building_has_goods_shortage = no
				}
				any_scope_building = {
					is_building_type = building_arms_industry
					level >= 3
					occupancy >= 0.9
					building_has_goods_shortage = no
				}
			}
		}
		trigger_if = {
			limit = {
				is_ai = yes
			}
			army_size >= 100
			any_scope_state = {
				state_region = s:STATE_URGA
				any_scope_building = {
					is_building_type = building_university
					level >= 5
				}
				any_scope_building = {
					is_building_type = building_arms_industry
					level >= 3
				}
			}
		}
	}
	on_complete = {
		trigger_event = { id = xiong_nu_events.1 }
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					is_ai = yes
					var:ai_je_loong_city_revival < 60
				}
			    change_variable = {
					name = ai_je_loong_city_revival
					add = 1
				}
			}
			if = {
				limit = {
					is_ai = yes
					var:ai_je_loong_city_revival >= 60
					NOT = { has_variable = ai_complete_je_loong_city_revival }
				}
				set_variable = ai_complete_je_loong_city_revival
				random_scope_state = {
					limit = {
						state_region = s:STATE_URGA
					}
					create_building={
						building="building_university"
					    level >= 5
					}
					create_building={
						building="building_arms_industry"
					    level >= 3
					}
				}
			}
		}
	}
    weight = 1000
	should_be_pinned_by_default = yes
}
